  DEGREES_PER_MILE = 1 / 69.047

    stores = []
    if doc['ok'] == 1.0

        distance_in_degrees = result['dis']
        obj = result['obj']

        store['distance'] = Float(distance_in_degrees) * MILES_PER_DEGREE
        stores.push(store)
      end
    end
