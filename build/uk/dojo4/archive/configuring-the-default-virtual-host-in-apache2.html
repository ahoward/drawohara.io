```html Налаштування VirtualHost за замовчуванням в apache2
@drawohara
#
#
# apache має таку особливість, що будь-який запит з поганим або невідомим
# заголовком Host: спрямовується до першого віртуального хосту, виявленого в його конфігурації. це може спричинити
# серйозні проблеми, коли кілька сайтів розміщено на одному сервері apache.
# наприклад, невдалий запит для http://foo.bar.com/missing
# може потрапити на віртуальний хост passenger і спричинити невизначену поведінку в залежності
# від того, як поводиться це додаткове ПО.
#
# рішення полягає в тому, щоб забезпечити наявність віртуального хосту, який зловить все,
# і щоб цей віртуальний хост завантажувався перед усіма іншими файлами конфігурації віртуальних хостів - і щоб цей
# віртуальний хост просто повертав 404 помилку (ви також можете показати дружнє повідомлення про помилку).
#
# припустимо, що конфігураційні файли apache завантажуються з, наприклад:
#
# /etc/apache2/sites-enabled/*.conf
#
# ви повинні помістити цю конфігурацію в файл з назвою
#
# /etc/apache2/sites-enabled/0.conf
#
# '0.conf' забезпечує те, що файл завантажується першим. ви можете перевірити, що ця конфігурація
# завантажується першою, виконавши команду
#
# ls -1 /etc/apache2/sites-enabled/ | sort | head -1
#
# це повинно вивести
#
# /etc/apache2/sites-enabled/0.conf
#
# після внесення конфігурації ви можете перевірити, що вона справді є
# конфігурацією за замовчуванням, виконавши команду
#
# curl --silent --header 'Host: fu.bar.com' your.domain.com
#
# якщо все правильно, буде показана стандартна сторінка 404 помилки apache. звичайно,
# ви можете налаштувати віртуальний сервер за замовчуванням мати фактичний кореневий документ
# і т.д...
#
# ще одне важливе зауваження. недостатньо просто налаштувати
#
# _default_:80
#
# віртуальний хост.
#
# ви також повинні налаштувати фактичний названий хост
#
# ServerName _default_
#
# причина в тому, що віртуальне хостування на основі імен за замовчуванням використовує
# перший *віртуальний хост на основі імен*, тому ви не можете покладатися на хост без назви,
# що зазвичай достатньо в більшості ситуацій.
#
# цей розділ зловить запити до відсутніх хостів під конфігурацією віртуального хосту на основі імен
#
  <VirtualHost *:80>
    ServerName _default_
    Redirect 404 /
  </VirtualHost>
# а це оброблятиме решту
#
  <VirtualHost _default_:80>
    Redirect 404 /
  </VirtualHost>
```
