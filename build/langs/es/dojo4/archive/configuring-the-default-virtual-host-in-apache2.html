# apache tiene el comportamiento extraño de enrutar cualquier solicitud con un encabezado Host:
# malo o desconocido al primer host virtual encontrado en su config. esto puede causar
# problemas serios cuando múltiples sitios se alojan en la misma instancia de apache
# por ejemplo, una solicitud fallida para http://foo.bar.com/missing
# podría afectar a un host virtual de passenger y causar un comportamiento no especificado dependiendo de
# cómo se comporta esa aplicación.
#
# la solución es asegurarse de que exista un host virtual que lo capture todo y que se cargue
# antes de todos los demás archivos de configuración de host virtual - y que este host virtual simplemente
# devuelva un 404 (también podrías servir una página de error amigable)
#
# asumiendo un diseño donde los archivos de config de apache se cargan desde, por ejemplo:
#
# /etc/apache2/sites-enabled/*.conf
#
# deberías poner esta configuración en un archivo llamado
#
# /etc/apache2/sites-enabled/0.conf
#
# el '0.conf' asegura que se cargue primero. puedes validar que esta configuración
# se carga primero a través de
#
# ls -1 /etc/apache2/sites-enabled/ | sort | head -1
#
# esto debería imprimir
#
# /etc/apache2/sites-enabled/0.conf
#
# una vez que la configuración esté en su lugar, puedes probar que realmente es la
# configuración predeterminada/primera utilizada a través de
#
# curl --silent --header 'Host: fu.bar.com' tu.dominio.com
#
# la página 404 predeterminada de apache se servirá si esto es correcto. obviamente
# puedes configurar los servidores virtuales predeterminados para que tengan un documento raíz real, etc...
#
# otra nota importante. no es *suficiente* simplemente configurar el
#
# _default_:80
#
# host virtual.
#
# también necesitas configurar el host nombrado real
#
# ServerName _default_
#
# la razón de esto es que el alojamiento virtual basado en nombres usará, por defecto,
# el primer *host virtual basado en nombres* por lo que no puedes confiar en uno sin nombre,
# lo cual sería suficiente en la mayoría de las situaciones.
#
# esta sección capturará las solicitudes a hosts faltantes bajo una configuración de vhost basada en nombres
#
  <VirtualHost *:80>
    ServerName _default_
    Redirect 404 /
  </VirtualHost>
# y esto manejará el resto
#
  <VirtualHost _default_:80>
    Redirect 404 /
  </VirtualHost>
