def up
  #Vogliamo rinominare il campo libro.isbn in libro.numero_libro
  Collection.find({"isbn" => {"$exists" => 1}}).each do |book|
    update_op = {
      "$unset" => {"isbn" => 1},
      "$set" => {"numero_libro" => book["isbn"]}

end
