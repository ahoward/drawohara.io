# 1° latitud = 69.047 miles = 111.12 kilometers

  GRADER_PER_MILE = 1 / 69.047
  MILES_PER_GRAD = 1 / GRADER_PER_MILE
  def Butik.hitta_alla_efter_lat_lng(lat, lng, alternativ = {})
    alternativ.till_alternativ!
    miles = alternativ[:miles] || 100
    max_avstånd = miles * GRADER_PER_MILE
    dok = 
      Mongoid.session(:default).
        kommando(
          :geoNear     => Butik.samling_namn,
          :nära        => {:lat => lat, :lng => lng},
          :maxAvstånd => max_avstånd
        )
    butiker = []
    om dok['ok'] == 1.0
      resultat = Array(dok['resultat'])
      resultat.varje gör |resultat|
        avstånd_i_grader = resultat['dis']
        obj = resultat['obj']
        butik = Butik.instansiera(obj)
        butik['avstånd'] = Float(avstånd_i_grader) * MILES_PER_GRAD
        butiker.push(butik)
      slut
    slut
    butiker
  slut
