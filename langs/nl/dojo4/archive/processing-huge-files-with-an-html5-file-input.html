<script>
  jQuery('#bestand').change(function(){
    var bestand = document.getElementById('bestand').files[0];
    var voortgang = jQuery('#voortgang');
    if(bestand){
      var reader = new FileReader();
      var grootte = bestand.grootte;
      var chunk_grootte = Math.pow(2, 13);
      var chunks = [];
      var offset = 0;
      var bytes = 0;
      reader.onloadend = function(e){
        if (e.target.readyState == FileReader.DONE){
          var chunk = e.target.result;
          bytes += chunk.lengte;
          chunks.push(chunk);
          voortgang.html(chunks.lengte + ' chunks // ' + bytes + ' bytes...');
          if((offset < grootte)){
            offset += chunk_grootte;
            var blob = bestand.slice(offset, offset + chunk_grootte);
            reader.readAsText(blob);
          } else {
            voortgang.html("het verwerken van de inhoud...");
            var inhoud = chunks.join("");
            alert("inhoud is klaar!");
            debugger;
          };
        }
      };
      var blob = bestand.slice(offset, offset + chunk_grootte);
      reader.readAsText
