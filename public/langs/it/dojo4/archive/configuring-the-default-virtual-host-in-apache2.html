# apache ha il comportamento strano di instradare qualsiasi richiesta con un header Host:
# errato o sconosciuto al primo virtual host trovato nella sua configurazione. questo può causare
# problemi seri quando più siti sono ospitati nella stessa istanza di apache
# per esempio, una richiesta non riuscita per http://foo.bar.com/mancante
# potrebbe colpire un virtual host passenger e causare un comportamento non specificato a seconda di
# come l'applicazione si comporta.
#
# la soluzione è assicurarsi che esista un virtual host catch-all e che sia caricato
# prima di tutti gli altri file di configurazione dei virtual host - e che questo host virtuale
# semplicemente restituisca un 404 (potresti anche servire una pagina di errore amichevole)
#
# assumendo un layout dove i file di configurazione di apache vengono caricati da, ad esempio:
#
# /etc/apache2/sites-enabled/*.conf
#
# dovresti mettere questa configurazione in un file chiamato
#
# /etc/apache2/sites-enabled/0.conf
#
# il '0.conf' assicura che venga caricato per primo. puoi verificare che questa configurazione
# è caricata per prima tramite
#
# ls -1 /etc/apache2/sites-enabled/ | sort | head -1
#
# questo dovrebbe stampare
#
# /etc/apache2/sites-enabled/0.conf
#
# una volta che la configurazione è in atto puoi testare che sia effettivamente la
# configurazione predefinita/prima utilizzata tramite
#
# curl --silent --header 'Host: fu.bar.com' il.tuo.dominio.com
#
# la pagina 404 predefinita di apache verrà servita se questo è corretto. ovviamente
# puoi configurare i server virtuali predefiniti per avere un documento root effettivo, ecc...
#
# un'altra nota importante. non è *sufficiente* configurare semplicemente il
#
# _default_:80
#
# virtual host.
#
# è anche necessario impostare l'host denominato effettivo
#
# ServerName _default_
#
# il motivo di ciò è che l'hosting virtuale basato su nome utilizzerà, per impostazione predefinita,
# il primo *host virtuale basato su nome* quindi non puoi fare affidamento su uno senza nome,
# che sarebbe sufficiente nella maggior parte delle situazioni.
#
# questa sezione catturerà le richieste agli host mancanti sotto una configurazione di vhost basata su nome
#
  <VirtualHost *:80>
    ServerName _default_
    Redirect 404 /
  </VirtualHost>
# e questo gestirà il resto
#
  <VirtualHost _default_:80>
    Redirect 404 /
  </VirtualHost>
