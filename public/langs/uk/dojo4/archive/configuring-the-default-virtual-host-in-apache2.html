# apache має дивну поведінку переадресації будь-якого запиту з поганим або невідомим
# заголовком Host: до першого віртуального хоста, знайденого в його конфігурації. це може викликати
# дуже серйозні проблеми, коли кілька сайтів розміщено на одному інстансі apache
# наприклад, невдалий запит на http://foo.bar.com/missing
# може влучити в віртуальний хост passenger і викликати невизначену поведінку в залежності
# від поведінки цього додатку.
#
# рішення полягає в тому, щоб забезпечити наявність віртуального хоста, який ловить все, і щоб він
# завантажувався перед усіма іншими файлами конфігурації віртуальних хостів - і щоб цей віртуальний
# хост просто повертав 404 (ви також можете показувати дружнє повідомлення про помилку)
#
# припустимо, що конфігураційні файли apache завантажуються з, наприклад:
#
# /etc/apache2/sites-enabled/*.conf
#
# ви повинні помістити цю конфігурацію в файл з назвою
#
# /etc/apache2/sites-enabled/0.conf
#
# '0.conf' забезпечує його завантаження першим. ви можете перевірити, що ця конфігурація
# завантажується першою за допомогою
#
# ls -1 /etc/apache2/sites-enabled/ | sort | head -1
#
# це повинно надрукувати
#
# /etc/apache2/sites-enabled/0.conf
#
# після встановлення конфігурації ви можете перевірити, що вона дійсно є
# конфігурацією за замовчуванням/першою використовуваною конфігурацією за допомогою
#
# curl --silent --header 'Host: fu.bar.com' your.domain.com
#
# якщо це правильно, буде показана стандартна сторінка 404 apache. очевидно,
# ви можете налаштувати віртуальні сервери за замовчуванням так, щоб у них був фактичний кореневий документ
# і т.д...
#
# ще одна важлива заувага. недостатньо просто налаштувати
#
# _default_:80
#
# віртуальний хост.
#
# вам також потрібно налаштувати фактичний іменований хост
#
# ServerName _default_
#
# причина цього полягає в тому, що віртуальне хостинг на основі імені за замовчуванням використовує
# перший *віртуальний хост на основі імені*, тому ви не можете покладатися на неіменований,
# який був би достатнім у більшості ситуацій.
#
# цей розділ оброблятиме запити до відсутніх хостів під конфігурацією віртуального хоста на основі імені
#
#
  <VirtualHost *:80>
    ServerName _default_
    Redirect 404 /
  </VirtualHost>
# а це оброблятиме все інше
#
  <VirtualHost _default_:80>
    Redirect 404 /
  </VirtualHost>
