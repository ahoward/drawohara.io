# apache har det underliga beteendet att dirigera alla förfrågningar med en dålig eller okänd
# Host: header till den första virtuella värden som finns i dess konfiguration. Detta kan orsaka
# mycket allvarliga problem när flera webbplatser är värd för samma apache
# instans. Till exempel kan en misslyckad förfrågan för http://foo.bar.com/missing
# träffa en virtuell passagerarvärd och orsaka odefinierat beteende beroende på
# hur denna applikation beter sig.
#
# lösningen är att se till att en catch-all virtuell värd finns och laddas
# före alla andra virtuella värd konfigurationsfiler - och att denna virtuella
# värd enbart returnerar en 404-fel (du kan också servera en vänlig felmeddelandesida)
#
# antag en layout där apache konffiler laddas från, t.ex:
#
# /etc/apache2/sites-enabled/*.conf
#
# du bör placera denna konfiguration i en fil som heter
#
# /etc/apache2/sites-enabled/0.conf
#
# '0.conf' säkerställer att den laddas först. Du kan validera att denna konfiguration
# laddas först via
#
# ls -1 /etc/apache2/sites-enabled/ | sort | head -1
#
# detta bör skriva ut
#
# /etc/apache2/sites-enabled/0.conf
#
# när konfigurationen är på plats kan du testa att det verkligen är den
# standard/första konfigurationen som används via
#
# curl --silent --header 'Host: fu.bar.com' your.domain.com
#
# standard Apache 404-sidan kommer att serveras om detta är korrekt. Självklart
# kan du konfigurera standard virtuella servrar för att ha en faktisk dokument
# rot, o.s.v...
#
# en annan viktig anmärkning. Det är *inte* tillräckligt att enbart konfigurera
#
# _default_:80
#
# virtuell värd.
#
# du måste också ställa in den faktiska namngivna värden
#
# ServerName _default_
#
# anledningen till detta är att namnbaserad virtuell värdning kommer att använda, som standard,
# den första *namnbaserade virtuella värden* så du kan inte lita på en oanvänd,
# som skulle räcka i de flesta situationer.
#
# den här sektionen kommer att fånga förfrågningar till saknade värdar under en namnbaserad vhost
# konfiguration
#
  <VirtualHost *:80>
    ServerName _default_
    Redirect 404 /
  </VirtualHost>
# och detta kommer att hantera resten
#
  <VirtualHost _default_:80>
    Redirect 404 /
  </VirtualHost>
