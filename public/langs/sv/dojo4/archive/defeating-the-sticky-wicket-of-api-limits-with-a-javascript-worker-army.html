TL;DR

när du har mycket för mycket arbete att göra på servern med avseende på api-gränser

ge något av arbetet till dina klienters webbläsare att utföra - skicka resultaten tillbaka till din databas

problemet

förra veckan hade jag att göra med (g-provider) api-gränser och för många geolokalisationsförfrågningar som kom från en server(s). Jag arbetade lite med det, gjorde saker i bakgrunden, med periodiska försök - du vet: det vanliga…

det aktuella apiet är begränsat per ip, och dokumentationen rekommenderar att man tar fördel av detta genom att göra de flesta api-anrop på klientsidan - i js.

nu, i det här specifika fallet behövde jag absolut resultaten av api-anropen på serversidan, så jag kom på den här kompromisslösningen (på flera sätt).

lösningen

antag att vissa klienter är villiga att bränna sin cpu för att fylla på data de är intresserade av att arbeta med. På rent engelska, låt vissa av dina klienter köra jobb från en javascript-arbetskö för att pumpa den källa de är intresserade av att dricka från.

i mitt fall har jag vissa valda gränssnitt (de som är intresserade av utdata från sagda jobb) som kör en liten iframe

det js som krävs för att interagera med jobbkö-biblioteket är kort och enkelt

allt det gör är att få ett jobb, köra det och skicka det tillbaka till servern - respektera både genomströmning och max jobb att köra.

if(!window.jobs){
//
  window.jobs = {};
  window.jobs.count = 0;
  window.jobs.max = 256;
  window.jobs.throttle = 1000;
  window.jobs.complete = function(){};
//
  jobs.get_next_job = function(){
    var success = function(response){
      job = response['data']['job'];
      if(job){
        jobs.count++;
        jobs.run(job, function(job){
          if(jobs.count < jobs.max){
            setTimeout(jobs.get_next_job, jobs.throttle);
          }
        });
      }
    };
    jQuery.ajax({
      'url'     : '/api/jobs/next',
      'type'    : 'GET',
      'cache'   : false,
      'success' : success
    });
  };
//
  jobs.run = function(job, callback){
    var code = job['code'];
    var result = undefined;
    callback = callback || function(){};
    if(code){
      try{
        (function(){
          job['result'] = eval(code);
        })()
      } catch(e){};
    }
    var url = '/api/jobs/:id'.replace(/:id/, job['id']);
    var data = {'job' : job};
    var success = function(response){
      job = response['data']['job'];
      try{
        jobs.complete(job);
      } catch(e){};
      try{
        callback(job);
      } catch(e){};
    };
    jQuery.ajax({
      'url'      : url,
      'type'     : 'PUT',
      'cache'   : false,
      'data'     : {'job'  : job},
      'dataType' : 'json',
      'success'  : success
    });
  };
//
  jobs.start = jobs.get_next_job;
}
